using System.Collections;
using System.Collections.Generic;
using UnityEngine;

public class CreatCubeAlt : MonoBehaviour
{
    [SerializeField] List<GameObject> spawns;
    [SerializeField] GameObject objetoEnemigo;

    int index;

    // Start is called before the first frame update
    void Start()
    {
      // Desactivar la generación de objetos al inicio
        enabled = true;
    }

    // Método llamado cuando un objeto es destruido
    public void ObjectDestroyed()
    {
        // Activar la generación de objetos
        enabled = false;
    }
    // Método para crear un enemigo
    public void CreateEnemy()
    {

        index = Random.Range(0, spawns.Count);
        GameObject obj = Instantiate(objetoEnemigo, spawns[index].transform.position,
            spawns[index].transform.rotation);

        // Destruir el objeto después de 5 segundos
        Destroy(obj, 5f);
    }
}
